# EXCEL SHORTCUTS:

- CTRL+S #SAVE
- CTRL+B "BOLD FONT
- CTRL+I "ITALIC FONT
- CTRL+U #UNDELINE
- CTRL+X #CUT
- CTRL+C #COPY
- CTRL+V #PASTE
- ESC #DESELECT CELL
- CTRL+O #OPENNING OPEN PAGE
- CTRL+ARROWS #MOVE TO LAST COLUMN OR ROW
- CTRL+DOWN_ARROW #SELECT COLUMNS
- CTRL+RIGHT_ARROW #SELECT ROWS
- CTRL+A #SELECT_ALL_CELLS
- CTRL+P #OPENNING PRINT PAGE
- ALT #SHOW KEYBOARD SHORTCUTS
